### 1. Create Learning Callout Component
**File**: `src/components/LearningCallout.tsx` (new)

Show inline callouts when Opta uses learned preferences:

```tsx
interface LearningCalloutProps {
  preference: string;
  action: string;
  onDismiss?: () => void;
  onChangePreference?: () => void;
}

export function LearningCallout({
  preference,
  action,
  onDismiss,
  onChangePreference,
}: LearningCalloutProps) {
  return (
    <motion.div
      initial={{ opacity: 0, x: -10 }}
      animate={{ opacity: 1, x: 0 }}
      exit={{ opacity: 0, x: -10 }}
      className="glass-subtle rounded-lg p-3 flex items-start gap-3"
    >
      <Brain className="w-5 h-5 text-primary flex-shrink-0 mt-0.5" />
      <div className="flex-1">
        <p className="text-sm">
          <span className="text-primary font-medium">Based on your history:</span>{' '}
          {action}
        </p>
        <p className="text-xs text-muted-foreground mt-1">
          {preference}
        </p>
        <div className="flex gap-2 mt-2">
          <button
            onClick={onChangePreference}
            className="text-xs text-primary hover:underline"
          >
            Change this preference
          </button>
          <button
            onClick={onDismiss}
            className="text-xs text-muted-foreground hover:underline"
          >
            Don't show again
          </button>
        </div>
      </div>
    </motion.div>
  );
}
```

### 2. Create Monthly Learning Summary
**File**: `src/components/LearningSummary.tsx` (new)

```tsx
export function LearningSummary() {
  const { learnedPreferences, learningStats } = useLearning();

  return (
    <div className="glass rounded-xl p-6">
      <div className="flex items-center gap-2 mb-4">
        <Brain className="w-5 h-5 text-primary" />
        <h3 className="font-semibold">What Opta Learned This Month</h3>
      </div>

      <div className="space-y-3">
        {learnedPreferences.map((pref, i) => (
          <motion.div
            key={pref.id}
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: i * 0.1 }}
            className="flex items-center justify-between"
          >
            <div className="flex items-center gap-2">
              {pref.isPositive ? (
                <ThumbsUp className="w-4 h-4 text-success" />
              ) : (
                <ThumbsDown className="w-4 h-4 text-danger" />
              )}
              <span className="text-sm">{pref.description}</span>
            </div>
            <span className="text-xs text-muted-foreground">
              {pref.confidence}% confident
            </span>
          </motion.div>
        ))}
      </div>

      <div className="mt-4 pt-4 border-t border-border/30">
        <p className="text-sm text-muted-foreground">
          <span className="font-medium">{learningStats.totalDecisions}</span> decisions analyzed •{' '}
          <span className="font-medium">{learningStats.preferencesLearned}</span> preferences learned
        </p>
      </div>
    </div>
  );
}
```

### 3. Create Editable Preferences Panel
**File**: `src/components/EditablePreferences.tsx` (new)

```tsx
export function EditablePreferences() {
  const { learnedPreferences, updatePreference, deletePreference } = useLearning();

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <h3 className="font-semibold">Your Preferences</h3>
        <Button variant="outline" size="sm" onClick={() => deleteAllPreferences()}>
          <Trash2 className="w-4 h-4 mr-1" />
          Clear All
        </Button>
      </div>

      <div className="space-y-2">
        {learnedPreferences.map(pref => (
          <div key={pref.id} className="glass-subtle rounded-lg p-4">
            <div className="flex items-start justify-between">
              <div>
                <p className="font-medium">{pref.name}</p>
                <p className="text-sm text-muted-foreground">{pref.description}</p>
              </div>
              <div className="flex gap-1">
                <Button
                  variant="ghost"
                  size="sm"
                  onClick={() => updatePreference(pref.id, { enabled: !pref.enabled })}
                >
                  {pref.enabled ? (
                    <ToggleRight className="w-5 h-5 text-success" />
                  ) : (
                    <ToggleLeft className="w-5 h-5 text-muted-foreground" />
                  )}
                </Button>
                <Button
                  variant="ghost"
                  size="sm"
                  onClick={() => deletePreference(pref.id)}
                >
                  <X className="w-4 h-4" />
                </Button>
              </div>
            </div>

            {/* Adjustable slider for preference strength */}
            {pref.type === 'priority' && (
              <div className="mt-3">
                <label className="text-xs text-muted-foreground">Priority: {pref.value}%</label>
                <Slider
                  value={[pref.value]}
                  onValueChange={([v]) => updatePreference(pref.id, { value: v })}
                  min={0}
                  max={100}
                  step={10}
                  className="mt-1"
                />
              </div>
            )}
          </div>
        ))}

        {learnedPreferences.length === 0 && (
          <p className="text-center text-muted-foreground py-8">
            Opta will learn your preferences as you use the app.
          </p>
        )}
      </div>
    </div>
  );
}
```
