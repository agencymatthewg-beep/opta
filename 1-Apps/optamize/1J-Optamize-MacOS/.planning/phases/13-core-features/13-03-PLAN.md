### 3. Create Preset Manager Hook
**File**: `src/hooks/usePresets.ts` (new)

```tsx
export function usePresets() {
  const [presets, setPresets] = useState<OptimizationPreset[]>(() => {
    const saved = localStorage.getItem('opta-presets');
    const custom = saved ? JSON.parse(saved) : [];
    return [...DEFAULT_PRESETS, ...custom];
  });

  const [activePreset, setActivePreset] = useState<string | null>(() => {
    return localStorage.getItem('opta-active-preset');
  });

  const savePreset = (preset: Omit<OptimizationPreset, 'id' | 'isBuiltIn'>) => {
    const newPreset = {
      ...preset,
      id: `custom-${Date.now()}`,
      isBuiltIn: false,
    };
    const updated = [...presets, newPreset];
    setPresets(updated);
    localStorage.setItem('opta-presets', JSON.stringify(updated.filter(p => !p.isBuiltIn)));
  };

  const deletePreset = (id: string) => {
    const preset = presets.find(p => p.id === id);
    if (preset?.isBuiltIn) return;  // Can't delete built-in
    const updated = presets.filter(p => p.id !== id);
    setPresets(updated);
    localStorage.setItem('opta-presets', JSON.stringify(updated.filter(p => !p.isBuiltIn)));
  };

  const applyPreset = (id: string) => {
    setActivePreset(id);
    localStorage.setItem('opta-active-preset', id);
  };

  return { presets, activePreset, savePreset, deletePreset, applyPreset };
}
```

### 4. Create Preset Selector UI
**File**: `src/components/PresetSelector.tsx` (new)

```tsx
export function PresetSelector() {
  const { presets, activePreset, applyPreset } = usePresets();
  const icons = { Zap, Video, Volume2, Target, Sliders };

  return (
    <div className="grid grid-cols-2 gap-3">
      {presets.map(preset => {
        const Icon = icons[preset.icon] || Sliders;
        const isActive = activePreset === preset.id;

        return (
          <motion.button
            key={preset.id}
            onClick={() => applyPreset(preset.id)}
            className={cn(
              "glass-subtle rounded-lg p-4 text-left transition-all",
              isActive && "ring-2 ring-primary"
            )}
            whileHover={{ scale: 1.02 }}
            whileTap={{ scale: 0.98 }}
          >
            <Icon className={cn("w-5 h-5 mb-2", isActive ? "text-primary" : "text-muted-foreground")} />
            <p className="font-medium">{preset.name}</p>
            <p className="text-xs text-muted-foreground">{preset.description}</p>
          </motion.button>
        );
      })}
    </div>
  );
}
```
